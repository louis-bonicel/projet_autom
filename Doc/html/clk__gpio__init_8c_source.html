184 static void __attribute__((used))
185 clock_init (void)
186 {
187  /* HSI setup */
188  RCC->CR |= RCC_CR_HSION;
189  while (!(RCC->CR & RCC_CR_HSIRDY))
190  ;
191  /* Reset HSEON, HSEBYP, CSSON, and PLLON, not touching RCC_CR_HSITRIM */
192  RCC->CR &= (RCC_CR_HSITRIM | RCC_CR_HSION);
193  RCC->CFGR = 0;
194  while ((RCC->CFGR & RCC_CFGR_SWS) != RCC_CFGR_SWS_HSI)
195  ;
196 
197 #if !defined(MCU_STM32F0)
198  /* HSE setup */
199  RCC->CR |= RCC_CR_HSEON;
200  while (!(RCC->CR & RCC_CR_HSERDY))
201  ;
202 #endif
203 
204  /* PLL setup */
205  RCC->CFGR |= STM32_PLLMUL | STM32_PLLXTPRE | STM32_PLLSRC;
206  RCC->CR |= RCC_CR_PLLON;
207  while (!(RCC->CR & RCC_CR_PLLRDY))
208  ;
209 
210  /* Clock settings */
211  RCC->CFGR = STM32_MCOSEL | STM32_USBPRE | STM32_PLLMUL | STM32_PLLXTPRE
212  | STM32_PLLSRC | STM32_ADCPRE | STM32_PPRE2 | STM32_PPRE1 | STM32_HPRE;
213 
214  /*
215  * We don't touch RCC->CR2, RCC->CFGR2, RCC->CFGR3, and RCC->CIR.
216  */
217 
218  /* Flash setup */
219  FLASH->ACR = STM32_FLASHBITS;
220 
221  /* CRC */
222  RCC->AHBENR |= RCC_AHBENR_CRCEN;
223 
224  /* Switching on the configured clock source. */
225  RCC->CFGR |= STM32_SW;
226  while ((RCC->CFGR & RCC_CFGR_SWS) != (STM32_SW << 2))
227  ;
228 
229 #if defined(MCU_STM32F0)
230  RCC->APB2ENR |= RCC_APB2ENR_SYSCFGEN;
231  RCC->APB2RSTR = RCC_APB2RSTR_SYSCFGRST;
232  RCC->APB2RSTR = 0;
233 
234 # if defined(HAVE_SYS_H)
235  /* Use vectors on RAM */
236  SYSCFG->CFGR1 = (SYSCFG->CFGR1 & ~SYSCFG_CFGR1_MEM_MODE) | 3;
237 # endif
238 #endif
239 }
