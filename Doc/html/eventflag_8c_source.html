56 eventflag_wait (struct eventflag *ev)
57 {
58  int n;
59 
60  if (ev->sleeper)
61  chx_fatal (EVENTFLAG_ERR_WAIT);
62 
63  chopstx_mutex_lock (&ev->mutex);
64  if (!ev->flag)
65  chopstx_cond_wait (&ev->u.cond, &ev->mutex);
66 
67  n = __builtin_ffs (ev->flag);
68  ev->flag &= ~(1 << (n - 1));
69  chopstx_mutex_unlock (&ev->mutex);
70 
71  return (1 << (n - 1));
72 }

92  }
93 
94  n = __builtin_ffs (ev->flag);
95  if (n)

97  em = (1 << (n - 1));

99  }
100 
101  chopstx_mutex_unlock (&ev->mutex);
102  return em;
