105  * This routine only changes PSP and not MSP.

107 STATIC_ENTRY __attribute__ ((naked,section(".text.startup.0")))
108 void entry (void)
109 {
110  asm volatile ("bl clock_init\n\t"
111  /* Clear BSS section. */
112  "mov r0, #0\n\t"
113  "ldr r1, =_bss_start\n\t"
114  "ldr r2, =_bss_end\n"
115  "0:\n\t"
116  "cmp r1, r2\n\t"
117  "beq 1f\n\t"
118 #if defined(__ARM_ARCH_6M__)
119  "str r0, [r1]\n\t"
120  "add r1, #4\n\t"
121 #else
122  "str r0, [r1], #4\n\t"
123 #endif
124  "b 0b\n"
125  "1:\n\t"
126  /* Copy data section. */
127  "ldr r1, =_data\n\t"
128  "ldr r2, =_edata\n\t"
129  "ldr r3, =_textdata\n"
130  "2:\n\t"
131  "cmp r1, r2\n\t"
132  "beq 3f\n\t"
133 #if defined(__ARM_ARCH_6M__)
134  "ldr r0, [r3]\n\t"
135  "str r0, [r1]\n\t"
136  "add r3, #4\n\t"
137  "add r1, #4\n\t"
138 #else
139  "ldr r0, [r3], #4\n\t"
140  "str r0, [r1], #4\n\t"
141 #endif
142  "b 2b\n"
143  "3:\n\t"
144  /* Switch to PSP. */
145  "ldr r0, =__process0_stack_end__\n\t"
146  COMPOSE_STATEMENT ("sub r0, #", CHOPSTX_THREAD_SIZE, "\n\t")
147  "msr PSP, r0\n\t" /* Process (main routine) stack. */
148  "mov r1, #2\n\t"
149  "msr CONTROL, r1\n\t"
150  "isb\n\t"
151  "bl chx_init\n\t"
152  "bl chx_systick_init\n\t"
153  "bl gpio_init\n\t"
154  /* Enable interrupts. */
155 #if defined(__ARM_ARCH_7M__)
156  "mov r0, #0\n\t"
157  "msr BASEPRI, r0\n\t"
158 #endif
159  "cpsie i\n\t"
160  /* Call main. */
161  "mov r1, r0\n\t"
162  "bl main\n"
163  "4:\n\t"
164  "b 4b"
165  : /* no output */ : /* no input */ : "memory");
166 }
