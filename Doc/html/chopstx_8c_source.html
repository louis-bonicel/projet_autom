473 static void
474 chx_timer_insert (struct chx_thread *tp, uint32_t usec)
475 {
476  struct chx_thread *q;
477  uint32_t ticks = usec_to_ticks (usec);
478  uint32_t next_ticks = chx_systick_get ();
479 
480  for (q = q_timer.next; q != (struct chx_thread *)&q_timer; q = q->next)
481  {
482  if (ticks < next_ticks)
483  {
484  ll_insert (tp, q);
485  chx_set_timer (tp->prev, ticks);
486  chx_set_timer (tp, (next_ticks - ticks));
487  break;
488  }
489  else
490  {
491  ticks -= next_ticks;
492  next_ticks = q->v;
493  }
494  }
495 
496  if (q == (struct chx_thread *)&q_timer)
497  {
498  ll_insert (tp, q);
499  chx_set_timer (tp->prev, ticks);
500  chx_set_timer (tp, 1); /* Non-zero for the last entry. */
501  }
502 }

507 {

580 void
581 chx_handle_intr (void)
582 {
583  chopstx_intr_t *intr;
584  register uint32_t irq_num;
585 
586  asm volatile ("mrs %0, IPSR\n\t"
587  "sub %0, #16" /* Exception # - 16 = interrupt number. */
588  : "=r" (irq_num) : /* no input */ : "memory");
589  chx_disable_intr (irq_num);
590  chx_spin_lock (&intr_lock);
591  for (intr = intr_top; intr; intr = intr->next)
592  if (intr->irq_num == irq_num)
593  break;
594 
595  if (intr)
596  {
597  intr->ready++;
598  if (intr->tp != running && intr->tp->state == THREAD_WAIT_INT)
599  {
600  chx_ready_enqueue (intr->tp);
601  chx_request_preemption (intr->tp->prio);
602  }
603  }
604  chx_spin_unlock (&intr_lock);
605 }

713  { /* should be one at most. */

719  }

723  if (running->flag_detached)
724  running->state = THREAD_FINISHED;
