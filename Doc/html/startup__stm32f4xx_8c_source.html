266 void Default_Reset_Handler(void)
267 {
268  /* Initialize data and bss */
269  unsigned long *pulSrc, *pulDest;
270 
271  /* Copy the data segment initializers from flash to SRAM */
272  pulSrc = &_sidata;
273 
274  for(pulDest = &_sdata; pulDest < &_edata; )
275  {
276  *(pulDest++) = *(pulSrc++);
277  }
278 
279  /* Zero fill the bss segment. This is done with inline assembly since this
280  will clear the value of pulDest if it is not kept in a register. */
281  __asm(" ldr r0, =_sbss\n"
282  " ldr r1, =_ebss\n"
283  " mov r2, #0\n"
284  " .thumb_func\n"
285  "zero_loop:\n"
286  " cmp r0, r1\n"
287  " it lt\n"
288  " strlt r2, [r0], #4\n"
289  " blt zero_loop");
290 #ifdef __FPU_USED
291  /* Enable FPU.*/
292  __asm(" LDR.W R0, =0xE000ED88\n"
293  " LDR R1, [R0]\n"
294  " ORR R1, R1, #(0xF << 20)\n"
295  " STR R1, [R0]");
296 #endif
297 
298  // SystemInit();
299  /* Call the application's entry point.*/
300  main();
301 }
